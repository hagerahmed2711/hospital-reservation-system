    private void checkdata(){
        flag1=0;
        flag2=0;
        flag3=0;
        flag4=0;
        flag5=0;
        flag6=0;
        flag7=0;
        flag8=0;        
        checkphonenumber.setText("");
        checkspecialization.setText("");
         checknationalid.setText("");
         checkpassword.setText("");
         checkname.setText("");
         checkaddress.setText("");
         checkusername.setText("");
        dpn= doctorphonenumber.getText();
        dnid= doctornationalid.getText();
        dg= genderlist.getSelectedItem().toString();
        ds= specializationlist.getSelectedItem().toString();
        dp= doctorpassword.getText();
        dn=doctorname.getText();
        daddress=doctoraddress.getText();
        du=doctorusername.getText();
        if(dpn.equals("")){
            checkphonenumber.setText("please enter a valid phone number");
            flag1=1;
        }
        if(ds.equals("select")){
            checkspecialization.setText("please select the doctor specialization");
            flag3=1;
        }
        
        if(dnid.equals("")){
            checknationalid.setText("please enter a valid national id");
            flag2=1;
        }
        
        if(dp.equals("")){
            checkpassword.setText("please choose a password");
            flag4=1;
        }
        if(dn.equals("")){
            checkname.setText("please enter doctor's name");
            flag5=1;
        }
        if(daddress.equals("")){
            checkaddress.setText("please enter doctor's address");
            flag6=1;
        }
        
        if(du.equals("")){
            checkusername.setText("please choose a username");
            flag7=1;
        }
        String url="jdbc:derby://localhost:1527/HOSPITALRESERVATIONSYSTEMDATABASE";
        String user="HAGER";
        String password="HAGER";
        String checku="";
         if(flag7==0){
        try{
            Connection con = DriverManager.getConnection(url, user, password);
            String s="SELECT DOCTORUSERNAME FROM HAGER.DOCTOR" ;
           
           Statement st= con.createStatement();
           ResultSet rs= null;
           rs=st.executeQuery(s);
           while(rs.next()){
             checku=  rs.getString("DOCTORUSERNAME");
               if(checku.equals(du)){
                   flag8=1;
               }
           }
           s="SELECT PATIENTUSERNAME FROM HAGER.PATIENT" ;
           rs=st.executeQuery(s);
           while(rs.next()){
             checku=  rs.getString("PATIENTUSERNAME");
               if(checku.equals(du)){
                   flag8=1;
               }
           }
           s="SELECT ADMINUSERNAME FROM HAGER.ADMIN" ;
           rs=st.executeQuery(s);
           while(rs.next()){
             checku=  rs.getString("ADMINUSERNAME");
               if(checku.equals(du)){
                   flag8=1;
               }
           }
           rs.close();
           st.close();
           con.close();
         
        }
        catch(SQLException e) {
            
            System.out.println("error");
        }
        }
         if(flag8==1){
             checkusername.setText("please choose another username");
         }
    }
